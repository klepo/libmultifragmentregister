307 cv::Point3_<double> PoseEstimator::intersect(cv::Point3_<double> n,
308  cv::Point3_<double> v0,
309  cv::Point3_<double> p0,
310  cv::Point3_<double> p1)
311 {
312  cv::Point3_<double> u = p1 - p0;
313  cv::Point3_<double> w = p0 - v0;
314  double D = n.dot(u);
315  double N = -n.dot(w);
316  double sI = N / D;
317  cv::Point3_<double> I = p0 + sI*u;
318  return I;
319 }
