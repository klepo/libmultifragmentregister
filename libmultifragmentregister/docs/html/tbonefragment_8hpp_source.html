134 dlib::matrix<float> TBoneFragment<MetricType>::
135 poseGradient(setPoseType setPose,
136  getPoseType getPose)
137 {
138  //qDebug() << "pose gradient";
139  assert(myValuesCount > 0);
140  const QVector3D pose = (this->*getPose)();
141 
142  const int ParamCount = 3;
143  myPlusPoints.resize(ParamCount);
144  myMinusPoints.resize(ParamCount);
145  //const float eps = 0.1;
146  const float eps = myPoseEps;
147  for(int i = 0; i < myXRayViews.size(); i++)
148  myXRayViews[i]->resizeVertexMasks(ParamCount);
149 
150  // Toto alokovat uz v momente, kdy je znama velikost myValuesCount
151  dlib::matrix<float> result(myValuesCount, ParamCount);
152 
153  float * vertices = NULL;
154  const int vn = myRenderers.at(0)->getMesh()->getNumberOfVertices();
155 
156  for(int col = 0; col < ParamCount; col++)
157  {
158  //qDebug() << "pose param" << col;
159  QVector3D plus(pose);
160  QVector3D minus(pose);
161 
162  plus[col] += eps;
163  minus[col] -= eps;
164 
165  (this->*setPose)(plus);
166  int row = 0;
167  for(int i = 0; i < myRenderers.size(); i++)
168  {
169  if(myObserver != NULL)
170  myObserver->beforeRendering();
171 
172  myRenderers[i]->renderNow();
173  if(vertices == NULL)
174  {
175  myXRayViews[0]->getRenderer()->getRecomputedVertices(vertices);
176  }
177 
178  if(myObserver != NULL)
179  myObserver->afterRendering();
180 
181  myXRayViews[i]->myPlusMasks[col] = myRenderers[i]->getVerticesMask(vertices, vn, myXRayViews[i]->getOpenGLCrop(), myXRayViews[i]->getAngle());
182  myPlusPoints[col] = getTransformedPoints();
183 
184  // toto zkusit predavat jako floatove pole, ne jako vektor
185  float * v = myXRayViews[i]->getMetric()->getValues();
186  for(int j = 0; j < myXRayViews[i]->getMetric()->valuesCount(); j++)
187  result(row++, col) = v[j];
188  }
189 
190  (this->*setPose)(minus);
191  row = 0;
192  for(int i = 0; i < myRenderers.size(); i++)
193  {
194  if(myObserver != NULL)
195  myObserver->beforeRendering();
196 
197  myRenderers[i]->renderNow();
198 
199  if(myObserver != NULL)
200  myObserver->afterRendering();
201 
202  myXRayViews[i]->myMinusMasks[col] = myRenderers[i]->getVerticesMask(vertices, vn, myXRayViews[i]->getOpenGLCrop(), myXRayViews[i]->getAngle());
203  myMinusPoints[col] = getTransformedPoints();
204 
205  // toto zkusit predavat jako floatove pole, ne jako vektor
206  float * v = myXRayViews[i]->getMetric()->getValues();
207  for(int j = 0; j < myXRayViews[i]->getMetric()->valuesCount(); j++)
208  {
209  result(row, col) = (result(row, col) - v[j]) / (2 * eps);
210  row += 1;
211  }
212  }
213  }
214 
215  delete vertices;
216 
217  (this->*setPose)(pose);
218  return result;
219 }
